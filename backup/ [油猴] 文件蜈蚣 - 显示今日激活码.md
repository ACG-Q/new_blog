// ==UserScript==
// @name         显示今日激活码
// @namespace    Violentmonkey Scripts
// @match        *://filecxx.com/zh_CN/activation_code.html*
// @grant        none
// @version      1.1
// @author       六记
// @description  显示今日有效的激活码，并提供复制功能
// @icon         https://filecxx.com/favicon.ico
// @require      https://code.jquery.com/jquery-3.7.1.min.js
// ==/UserScript==

(function ($) {
    // 函数：复制文本到剪贴板
    function copyToClipboard(text) {
        if (navigator.clipboard && window.isSecureContext) {
            return navigator.clipboard.writeText(text).then(function () {
                console.log("复制成功");
            }).catch(function (err) {
                console.error("复制失败：", err);
                return Promise.reject(err);
            });
        } else {
            return new Promise((resolve, reject) => {
                // 回退到使用 document.execCommand
                const textarea = document.createElement("textarea");
                textarea.value = text;
                document.body.appendChild(textarea);
                textarea.select();
                const success = document.execCommand("copy");
                document.body.removeChild(textarea);
                success ? resolve() : reject("复制失败");
            });
        }
    }

    // 函数：显示自定义提示框
    function showCustomAlert(message, success = true) {
        const alertBox = $("<div>").css({
            position: "fixed",
            top: "20px",
            right: "20px",
            padding: "10px 20px",
            background: success ? "#44c767" : "#e74c3c",
            color: "#fff",
            "border-radius": "5px",
            "box-shadow": "0px 0px 10px rgba(0, 0, 0, 0.1)",
            "z-index": "1000"
        }).text(message);

        $("body").append(alertBox);

        setTimeout(function () {
            alertBox.fadeOut(function () {
                $(this).remove();
            });
        }, 3000);
    }

    $(document).ready(function () {
        // 使用 jQuery 选择元素
        var codeTimes = $("#code_list > div.code_time");
        var codeTexts = $("#code_list > div.code_text");

        // 遍历激活码日期
        codeTimes.each(function (i) {
            var startTime = $(this).text().split(" - ")[0];
            var endTime = $(this).text().split(" - ")[1];
            var startf = new Date(startTime) <= new Date();
            var endf = new Date(endTime) >= new Date();

            if (!startf || !endf) {
                // 隐藏不在今日日期范围内的激活码
                $(this).hide();
                $(this).next(".code_text").hide();
                return;
            }

            // 创建复制按钮并绑定点击事件
            var copyButton = $("<button>复制</button>").css({
                padding: "5px 10px",
                "font-size": "14px",
                color: "#fff",
                background: "#007bff",
                border: "none",
                "border-radius": "5px",
                cursor: "pointer",
                "margin-left": "10px",
                "transition": "background 0.3s"
            }).hover(
                function () {
                    $(this).css("background", "#0056b3");
                },
                function () {
                    $(this).css("background", "#007bff");
                }
            ).click(function () {
                const button = $(this);
                button.prop("disabled", true).text("复制中...");

                // 复制文本到剪贴板
                copyToClipboard(codeTexts.eq(i).text())
                    .then(() => {
                        showCustomAlert("激活码已复制到剪贴板！");
                    })
                    .catch(() => {
                        showCustomAlert("复制失败，请重试！", false);
                    })
                    .finally(() => {
                        button.prop("disabled", false).text("复制");
                    });
            });

            // 将复制按钮添加到页面上
            $(this).append(copyButton);
        });
    });
})(jQuery);
