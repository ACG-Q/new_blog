```Gmeek-html
<div id="root"></div>
    <script>
        // GitHub页面的URL
        const pageUrl = 'https://github.com/ACG-Q/qd-templates/blob/main/LIST.md';

        // 使用fetch请求GitHub页面内容
        fetch(pageUrl)
            .then(response => {
                if (!response.ok) {
                    throw new Error('网络响应不是OK');
                }
                return response.text();
            })
            .then(htmlContent => {
                // 创建一个DOMParser对象来解析HTML内容
                const parser = new DOMParser();
                const doc = parser.parseFromString(htmlContent, 'text/html');

                // 提取article.markdown-body元素的内容
                const contentElement = doc.querySelector('article.markdown-body');
                if (contentElement) {
                    // 将内容设置到root元素
                    document.getElementById('root').innerHTML = contentElement.innerHTML;
                } else {
                    console.error('找不到article.markdown-body元素');
                }
            })
            .catch(error => {
                console.error('请求或处理页面内容时出错:', error);
            });
    </script>
    ```
